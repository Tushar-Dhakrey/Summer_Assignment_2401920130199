class Solution {
    public int firstUniqChar(String s) {
        int[] fre = new int[26];
        char[] chars = s.toCharArray();
        for(char c:chars){
            fre[c-'a']++;
        }
        for(int i=0;i<chars.length;i++){
            if(fre[chars[i]-'a']==1){
                return i;
            }
        }
        return -1;
    }
}
