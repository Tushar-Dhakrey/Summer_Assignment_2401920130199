class Solution {
    public int lengthOfLongestSubstring(String s) {
        int fre[] = new int[128];
        int left =0;
        int max = 0;
        int n = s.length();
        for(int right=0;right<n;right++){
            fre[s.charAt(right)]++;
            while(fre[s.charAt(right)]>1){
                fre[s.charAt(left)]--;
                left++;
            }
            int current = right - left +1;
            max = Math.max(max,current);
        }
        return max;
    }
}
