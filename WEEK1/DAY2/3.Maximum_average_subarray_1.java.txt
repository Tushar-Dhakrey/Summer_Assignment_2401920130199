class Solution {
    public double findMaxAverage(int[] nums, int k) {
        int sum = 0;
        int n = nums.length;
        for(int i=0;i<k;i++){
            sum += nums[i];
        }
        int maxsum = sum ;
        int left = 0;
        int right = k;
        while(right<n){
            sum -= nums[left];
            left ++;
            sum += nums[right];
            right++;
            maxsum = Math.max(maxsum,sum);
        }
        return (double)maxsum/k;
    }
}
